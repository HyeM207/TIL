class LinkedListStack:

		self.data = DoublyLinkedList()

		return self.data.getLength()

def splitTokens(exprStr):
    tokens = []
    val = 0
    valProcessing = False
    for c in exprStr:
        if c == ' ':
            continue
        if c in '0123456789':
            val = val * 10 + int(c)
            valProcessing = True
        else:
            if valProcessing:
                tokens.append(val)
                val = 0
            valProcessing = False
            tokens.append(c)
    if valProcessing:
        tokens.append(val)

    return tokens

def postfixEval(tokenList):
    valStack = ArrayStack()

    for token in tokenList :
        if type(token) is int : 
            valStack.push(token)
        elif token == "*":
            p1 = valStack.pop()
            p2 = valStack.pop()
            valStack.push(p2*p1)
        elif token == "/":
            p1 = valStack.pop()
            p2 = valStack.pop()
            valStack.push(p2/p1)
        elif token == '+':
            p1 = valStack.pop()
            p2 = valStack.pop()
            valStack.push(p2+p1)
        elif token == "-":
            p1 = valStack.pop()
            p2 = valStack.pop()
            valStack.push(p2-p1)
        
    return valStack.pop()


def solution(expr):
    tokens = splitTokens(expr)
    postfix = infixToPostfix(tokens)
    val = postfixEval(postfix)
